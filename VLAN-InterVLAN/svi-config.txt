
! SVIs and routed uplink on SW2 (multilayer switch)
enable
configure terminal

! Enable IP routing on multilayer switch
ip routing

! SVI for VLAN10 (last usable in 10.0.0.0/26 = 10.0.0.62)
interface Vlan10
 description SVI VLAN 10 - SALES - last usable gateway
 ip address 10.0.0.62 255.255.255.192
 no shutdown
exit

! SVI for VLAN20 (last usable in 10.0.0.64/26 = 10.0.0.126)
interface Vlan20
 description SVI VLAN 20 - SERVICES - last usable gateway
 ip address 10.0.0.126 255.255.255.192
 no shutdown
exit

! SVI for VLAN30 (last usable in 10.0.0.128/26 = 10.0.0.190)
interface Vlan30
 description SVI VLAN 30 - IT - last usable gateway
 ip address 10.0.0.190 255.255.255.192
 no shutdown
exit
ip default-gateway 10.0.0.194 
! Access ports on SW2 (hosts connected directly to SW2)
interface GigabitEthernet1/0/5
 description PC - VLAN10
 switchport mode access
 switchport access vlan 10
 no shutdown
exit

interface GigabitEthernet1/0/4
 description PC - VLAN10
 switchport mode access
 switchport access vlan 10
 no shutdown
exit

interface GigabitEthernet1/0/3
 description PC - VLAN20
 switchport mode access
 switchport access vlan 20
 no shutdown
exit

! Routed uplink to Router R1 (10.0.0.192/30)
interface GigabitEthernet1/0/2
 description Routed link to Router R1
 no switchport
 ip address 10.0.0.193 255.255.255.252
 no shutdown
exit

end
write memory
